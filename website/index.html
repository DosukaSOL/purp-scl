// A complete Solana program in Purp
program TokenVault {

  account Vault {
    owner: pubkey,
    balance: u64,
    name: string
  }

  pub instruction deposit(
    #[mut] signer user,
    #[mut] account vault,
    amount: u64
  ) {
    assert(vault.owner == user, "Not owner");
    vault.balance += amount;
    emit Deposited(user, amount);
  }

  event Deposited { user: pubkey, amount: u64 }
}
